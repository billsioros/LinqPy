    def __call__(self, source: Iterable[T]) -> Iterable[T]:
    """Filters out duplicate elements from the source sequence.

    Args:
        source (Iterable[T]): The source sequence.

    Yields:
        T: The unique elements from the source sequence.
    """
    seen = set()
    for item in source:
        if item not in seen:
            seen.add(item)
            yield item

            yield item
